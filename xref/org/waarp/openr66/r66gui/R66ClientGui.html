420     /**
421      * @author Frederic Bregier
422      *
423      */
424     public class R66ClientGuiActions extends SwingWorker<String, Integer> {
425         static final int CHECKCONNECTION = 1;
426         static final int STARTTRANSFER = 2;
427         static final int FILESELECT = 3;
428         int method;
429 
430         R66ClientGuiActions(int method) {
431             this.method = method;
432         }
433 
434         @Override
435         protected String doInBackground() throws Exception {
436             disableAllButtons();
437             startRequest();
438             switch (method) {
439                 case CHECKCONNECTION:
440                     checkConnection();
441                     break;
442                 case STARTTRANSFER:
443                     startTransfer();
444                     break;
445                 case FILESELECT:
446                     findFile();
447                     break;
448                 default:
449                     environnement.GuiResultat = Messages.getString("R66ClientGui.24"); //$NON-NLS-1$
450             }
451             setStatus(environnement.GuiResultat);
452             if (method != FILESELECT) {
453                 showDialog();
454             } else {
455                 enableAllButtons();
456             }
457             stopRequest();
458             return environnement.GuiResultat;
459         }
460     }
461 
462     private void showDialog() {
463         disableAllButtons();
464         if (dialog != null) {
465             dialog.dispose();
466             dialog = null;
467         }
468         if (dialog == null) {
469             dialog = new R66Dialog();
470             dialog.setLocationRelativeTo(frmRClientGui);
471             if (dialog.isAlwaysOnTopSupported()) {
472                 dialog.setAlwaysOnTop(true);
473             } else {
474                 dialog.toFront();
475             }
476         }
477         dialog.textPaneDialog.setText(environnement.GuiResultat);
478         dialog.setVisible(true);
479         dialog.requestFocus();
480     }

501     private void checkConnection() {
502         startRequest();
503         disableAllButtons();
504         environnement.hostId = (String) comboBoxHosts.getSelectedItem();
505         if (environnement.hostId == null || environnement.hostId.trim().isEmpty()) {
506             environnement.hostId = Messages.getString("R66ClientGui.26"); //$NON-NLS-1$
507             environnement.GuiResultat = Messages.getString("R66ClientGui.27"); //$NON-NLS-1$
508         } else {
509             environnement.hostId = environnement.hostId.trim();
510             environnement.checkConnection();
511         }
512         setStatus(environnement.GuiResultat);
513         showDialog();
514         stopRequest();
515     }
516 
517     private void startTransfer() {
518         logger.debug("start startTransfer");
519         disableAllButtons();
520         environnement.hostId = (String) comboBoxHosts.getSelectedItem();
521         environnement.ruleId = (String) comboBoxRules.getSelectedItem();
522         environnement.filePath = textFieldFile.getText();
523         environnement.information = textFieldInformation.getText();
524         environnement.isMD5 = checkBoxMD5.isSelected();
525 
526         boolean ok = true;
527         if (environnement.hostId == null || environnement.hostId.trim().isEmpty()) {
528             environnement.hostId = Messages.getString("R66ClientGui.26"); //$NON-NLS-1$
529             ok = false;
530         } else {
531             environnement.hostId = environnement.hostId.trim();
532         }
533         if (environnement.filePath == null || environnement.filePath.trim().isEmpty()) {
534             environnement.filePath = Messages.getString("R66ClientGui.30"); //$NON-NLS-1$
535             ok = false;
536         } else {
537             environnement.filePath = environnement.filePath.trim();
538         }
539         if (environnement.information == null || environnement.information.trim().isEmpty()) {
540             environnement.information = "";
541         } else {
542             environnement.information = environnement.information.trim();
543         }
544         if (environnement.ruleId == null || environnement.ruleId.trim().isEmpty()) {
545             environnement.ruleId = Messages.getString("R66ClientGui.32"); //$NON-NLS-1$
546             ok = false;
547         } else {
548             environnement.ruleId = environnement.ruleId.trim();
549         }
550         if (ok) {
551             logger.debug("start startTransfer: " + environnement.toString());
552             environnement.startsTransfer(progressBarTransfer, textFieldStatus);
553         } else {
554             environnement.GuiResultat = Messages.getString("R66ClientGui.34") + //$NON-NLS-1$
555                     Messages.getString("R66ClientGui.35") + environnement.hostId + Messages.getString("R66ClientGui.36") + environnement.ruleId + //$NON-NLS-1$ //$NON-NLS-2$
556                     Messages.getString("R66ClientGui.37") + environnement.filePath; //$NON-NLS-1$
557         }
558         setStatus(environnement.GuiResultat);
559         showDialog();
560     }

562     private void findFile() {
563         startRequest();
564         disableAllButtons();
565         try {
566             JFileChooser fc = new JFileChooser();
567             int returnVal = fc.showOpenDialog(frmRClientGui);
568             if (returnVal == JFileChooser.APPROVE_OPTION) {
569                 File file = fc.getSelectedFile();
570                 try {
571                     textFieldFile.setText(file.getCanonicalPath());
572                 } catch (IOException e) {
573                 }
574                 setFindFile();
575                 environnement.GuiResultat = Messages.getString("R66ClientGui.38"); //$NON-NLS-1$
576             }
577         } finally {
578             enableAllButtons();
579             stopRequest();
580         }
581     }

583     private void setFindFile() {
584         String text = null;
585         try {
586             text = textFieldFile.getText();
587         } catch (NullPointerException e1) {
588             text = null;
589         }
590         if (text != null) {
591             File file = new File(text);
592             if (file.exists()) {
593                 text = file.toURI().toString();
594             } else {
595                 //text = "unknown file";
596             }
597             textFieldFile.setText(text);
598         }
599     }
