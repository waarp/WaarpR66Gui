78      public void initialize(String[] args) {
79          WaarpLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
80          initLog();
81          if (args.length < 1) {
82              System.err.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
83              System.exit(2);
84          }
85          if (!FileBasedConfiguration
86                  .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
87              logger
88                      .error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
89              if (DbConstant.admin != null && DbConstant.admin.isActive()) {
90                  DbConstant.admin.close();
91              }
92              ChannelUtils.stopLogger();
93              System.exit(2);
94          }
95          Configuration.configuration.pipelineInit();
96          networkTransaction = new NetworkTransaction();
97      }

102             networkTransaction = null;
103         }

116     public boolean checkConnection() {
117         R66Future result = new R66Future(true);
118         TestPacket packet = new TestPacket("MSG", "TestConnection", 100);
119         Message transaction = new Message(networkTransaction, result,
120                 hostId, packet);
121         transaction.run();
122         result.awaitUninterruptibly(30000);
123         if (result.isSuccess()) {
124             R66Result r66result = result.getResult();
125             ValidPacket info = (ValidPacket) r66result.getOther();
126             GuiResultat = Messages.getString("R66Environment.2") + //$NON-NLS-1$
127                     info.getSheader();
128         } else {
129             GuiResultat = Messages.getString("R66Environment.3") + //$NON-NLS-1$
130                     result.getResult().toString();
131         }
132         return result.isSuccess();
133     }

135     public boolean startsTransfer(JProgressBar progressBar, JEditorPane textFieldStatus) {
136         logger.debug("start startTransfer2");
137         long time1 = System.currentTimeMillis();
138         R66Future future = new R66Future(true);
139         ProgressDirectTransfer transaction = new ProgressDirectTransfer(future, hostId,
140                 filePath, ruleId, information, isMD5, Configuration.configuration.getBLOCKSIZE(),
141                 DbConstant.ILLEGALVALUE, networkTransaction, 500,
142                 progressBar, textFieldStatus);
143         logger.info("Launch transfer: " + hostId + ":" + ruleId + ":" + filePath);
144         transaction.run();
145         future.awaitUninterruptibly();
146         progressBar.setIndeterminate(true);
147         progressBar.setValue(0);
148         progressBar.setVisible(false);
149         long time2 = System.currentTimeMillis();
150         long delay = time2 - time1;
151         R66Result result = future.getResult();
152         if (future.isSuccess()) {
153             if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
154                 GuiResultat = Messages.getString("R66Environment.8") + //$NON-NLS-1$
155                         result.getRunner().toShortNoHtmlString("<br>")
156                         +
157                         Messages.getString("R66Environment.10") + //$NON-NLS-1$
158                         hostId
159                         +
160                         (result.getFile() != null ? result.getFile().toString() + "" : Messages.getString("R66ClientGui.30")) + Messages.getString("R66Environment.13") + //$NON-NLS-2$ //$NON-NLS-3$
161                         delay;
162             } else {
163                 GuiResultat = Messages.getString("R66Environment.14") + //$NON-NLS-1$
164                         result.getRunner().toShortNoHtmlString("<br>")
165                         +
166                         Messages.getString("R66Environment.10") + //$NON-NLS-1$
167                         hostId
168                         +
169                         (result.getFile() != null ? result.getFile().toString() + "" : Messages.getString("R66ClientGui.30")) + Messages.getString("R66Environment.13") + //$NON-NLS-2$ //$NON-NLS-3$
170                         delay;
171             }
172         } else {
173             if (result == null || result.getRunner() == null) {
174                 GuiResultat = Messages.getString("R66Environment.20") + //$NON-NLS-1$
175                         Messages.getString("R66Environment.10") + //$NON-NLS-1$
176                         hostId + "     " + future
177                                 .getCause().getMessage();
178             } else if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
179                 GuiResultat = Messages.getString("R66Environment.23") + //$NON-NLS-1$
180                         result.getRunner().toShortNoHtmlString("<br>") +
181                         Messages.getString("R66Environment.10") + //$NON-NLS-1$
182                         hostId + "    " + future.getCause().getMessage();
183             } else {
184                 GuiResultat = Messages.getString("R66Environment.27") + //$NON-NLS-1$
185                         result.getRunner().toShortNoHtmlString("<br>") +
186                         Messages.getString("R66Environment.10") + //$NON-NLS-1$
187                         hostId + "    " + future.getCause().getMessage();
188             }
189         }
190         return future.isSuccess();
191     }

193     public static String[] getHostIds() {
194         String[] results = null;
195         DbHostAuth[] dbHostAuths;
196         try {
197             dbHostAuths = DbHostAuth.getAllHosts();
198         } catch (WaarpDatabaseNoConnectionException e) {
199             results = new String[1];
200             results[0] = Messages.getString("R66Environment.31"); //$NON-NLS-1$
201             return results;
202         }
203         if (dbHostAuths.length == 0) {
204             results = new String[1];
205             results[0] = Messages.getString("R66Environment.31"); //$NON-NLS-1$
206             return results;
207         }
208         results = new String[dbHostAuths.length];
209         for (int i = 0; i < dbHostAuths.length; i++) {
210             results[i] = dbHostAuths[i].getHostid();
211         }
212         return results;
213     }

237     public static String[] getRules(boolean sendMode) {
238         String[] results = null;
239         DbRule[] dbRules;
240         try {
241             dbRules = DbRule.getAllRules();
242         } catch (WaarpDatabaseNoConnectionException e) {
243             results = new String[1];
244             results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
245             return results;
246         }
247         int len = 0;
248         for (DbRule rule : dbRules) {
249             if (sendMode) {
250                 if (rule.isSendMode())
251                     len++;
252             } else {
253                 if (rule.isRecvMode())
254                     len++;
255             }
256         }
257         if (len == 0) {
258             results = new String[1];
259             results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
260             return results;
261         }
262         results = new String[len];
263         int i = 0;
264         for (DbRule rule : dbRules) {
265             if (sendMode) {
266                 if (rule.isSendMode()) {
267                     results[i] = rule.getIdRule();
268                     i++;
269                 }
270             } else {
271                 if (rule.isRecvMode()) {
272                     results[i] = rule.getIdRule();
273                     i++;
274                 }
275             }
276         }
277         return results;
278     }
