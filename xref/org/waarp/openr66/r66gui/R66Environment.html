82      public void initialize(String []args) {
83          InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
84          initLog();
85          if (args.length < 1) {
86              System.err.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
87              System.exit(2);
88          }
89          if (! FileBasedConfiguration
90                  .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
91              logger
92                      .error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
93              if (DbConstant.admin != null && DbConstant.admin.isConnected) {
94                  DbConstant.admin.close();
95              }
96              ChannelUtils.stopLogger();
97              System.exit(2);
98          }
99          Configuration.configuration.pipelineInit();
100         networkTransaction = new NetworkTransaction();
101     }

119     public boolean checkConnection() {
120         R66Future result = new R66Future(true);
121         TestPacket packet = new TestPacket("MSG", "TestConnection", 100);
122         Message transaction = new Message(networkTransaction, result,
123                 hostId, packet);
124         transaction.run();
125         result.awaitUninterruptibly(30000);
126         if (result.isSuccess()) {
127             R66Result r66result = result.getResult();
128             ValidPacket info = (ValidPacket) r66result.other;
129             GuiResultat = Messages.getString("R66Environment.2") + //$NON-NLS-1$
130                     info.getSheader();
131         } else {
132             GuiResultat = Messages.getString("R66Environment.3") + //$NON-NLS-1$
133                     result.getResult().toString();
134         }
135         return result.isSuccess();
136     }

147         transaction.run();

193     public static String [] getHostIds() {
194         String []results = null;
195         DbHostAuth[] dbHostAuths;
196         DbSession session = DbConstant.admin != null ? DbConstant.admin.session : null;
197         try {
198             dbHostAuths = DbHostAuth.getAllHosts(session);
199         } catch (WaarpDatabaseNoConnectionException e) {
200             results = new String[1];
201             results[0] = Messages.getString("R66Environment.31"); //$NON-NLS-1$
202             return results;
203         } catch (WaarpDatabaseSqlException e) {
204             results = new String[1];
205             results[0] = Messages.getString("R66Environment.31"); //$NON-NLS-1$
206             return results;
207         }
208         if (dbHostAuths.length == 0) {
209             results = new String[1];
210             results[0] = Messages.getString("R66Environment.31"); //$NON-NLS-1$
211             return results;
212         }
213         results = new String[dbHostAuths.length];
214         for (int i = 0; i < dbHostAuths.length; i++) {
215             results[i] = dbHostAuths[i].getHostid();
216         }
217         return results;

245     public static String [] getRules(boolean sendMode) {
246         String []results = null;
247         DbRule[] dbRules;
248         DbSession session = DbConstant.admin != null ? DbConstant.admin.session : null;
249         try {
250             dbRules = DbRule.getAllRules(session);
251         } catch (WaarpDatabaseNoConnectionException e) {
252             results = new String[1];
253             results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
254             return results;
255         } catch (WaarpDatabaseSqlException e) {
256             results = new String[1];
257             results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
258             return results;
259         }
260         int len = 0;
261         for (DbRule rule : dbRules) {
262         	if (sendMode) {
263         		if (rule.isSendMode()) len++;
264         	} else {
265         		if (rule.isRecvMode()) len++;
266         	}
267         }
268         if (len == 0) {
269             results = new String[1];
270             results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
271             return results;
272         }
273         results = new String[len];
274         int i = 0;
275         for (DbRule rule : dbRules) {
276         	if (sendMode) {
277         		if (rule.isSendMode()) {
278                     results[i] = rule.idRule;
279                     i++;
280         		}
281         	} else {
282         		if (rule.isRecvMode()) {
283                     results[i] = rule.idRule;
284                     i++;
285         		}
286         	}
287         }
288         return results;
289     }
